def ldclump_snppos(xarr, pos_cutoff=100000):
    xclump = list()
    xsorted = np.sort(xarr)
    xlast = xsorted[0]
    xclump.append(xlast)
    for x in xsorted[1:]:
        if (x - xlast > pos_cutoff):
            xclump.append(x)
            xlast = x
    return np.array(xclump)
